///$tab Main
SET ThousandSep=',';
SET DecimalSep='.';
SET MoneyThousandSep=',';
SET MoneyDecimalSep='.';
SET MoneyFormat='$#,##0.00;($#,##0.00)';
SET TimeFormat='h:mm:ss TT';
SET DateFormat='M/D/YYYY';
SET TimestampFormat='M/D/YYYY h:mm:ss[.fff] TT';
SET MonthNames='Jan;Feb;Mar;Apr;May;Jun;Jul;Aug;Sep;Oct;Nov;Dec';
SET DayNames='Mon;Tue;Wed;Thu;Fri;Sat;Sun';

///$tab Includes
$(Include=..\qvc_runtime\qvc.qvs);
///$tab Use FileExists
TRACE *****Start of examples ***************;

// Try this file. Correctly quoted as literal. Should return true.
LET vFileExists = $(Qvc.FileExists('FileExists.qvw'));
TRACE $(vFileExists) Correctly quoted literal;

// Try this file but without the quotes. Should return false.
LET vFileExists = $(Qvc.FileExists(FileExists.qvw));
TRACE $(vFileExists) Literal missing quotes;

// Try using a variable. Should return true.
// Notice we don't use $() around the parameter. We let the expansion  happen in the function.
SET vFilename = FileExists.qvw;
LET vFileExists = $(Qvc.FileExists(vFilename));
TRACE $(vFileExists) Variable;

// Try using an expression. Should return true.
// The expression will be evaluated in FileExists(), not before the call. 
LET vFileExists = $(Qvc.FileExists(DocumentPath()));
TRACE $(vFileExists) Expression;

// Test a empty (zero byte) file. Should return true.
LET vFileExists = $(Qvc.FileExists('EmptyFile.txt'));
TRACE $(vFileExists) Empty File;

// Test a non-existent file. Should return false.
LET vFileExists = $(Qvc.FileExists('MissingFile.txt'));
TRACE $(vFileExists) Missing File;



TRACE **** End of examples ****************;
///$tab Cleanup
// Cleanup Qvc
CALL Qvc.Cleanup;
